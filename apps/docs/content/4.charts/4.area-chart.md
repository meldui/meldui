## Basic Usage

The `MeldAreaChart` component renders data series as filled areas beneath a line. It shares most configuration options with `MeldLineChart` but adds fill-specific properties like `fillOpacity` and `stacked`.

```vue
<script setup lang="ts">
import { MeldAreaChart } from '@meldui/charts-vue'
import type { MeldAreaChartConfig } from '@meldui/charts-vue'

const config: MeldAreaChartConfig = {
  series: [{ name: 'Sessions', data: [3200, 4100, 3800, 5200, 4900, 6100] }],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
}
</script>

<template>
  <MeldAreaChart title="Website Sessions" :config="config" />
</template>
```

## Multiple Series

Compare several datasets with overlapping areas:

```vue
<script setup lang="ts">
import { MeldAreaChart } from '@meldui/charts-vue'
import type { MeldAreaChartConfig } from '@meldui/charts-vue'

const config: MeldAreaChartConfig = {
  series: [
    { name: 'Organic', data: [1200, 1500, 1800, 2200, 2600, 3000] },
    { name: 'Paid', data: [800, 1100, 1400, 1700, 2000, 2300] },
    { name: 'Referral', data: [400, 500, 600, 800, 900, 1100] },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  fillOpacity: 0.3,
  legend: { show: true, position: 'top' },
  tooltip: { shared: true },
}
</script>

<template>
  <MeldAreaChart title="Traffic Sources" :config="config" :height="400" />
</template>
```

## Stacked Area

Set `stacked: true` to stack areas on top of each other, showing the total and each series' contribution:

```vue
<script setup lang="ts">
import { MeldAreaChart } from '@meldui/charts-vue'
import type { MeldAreaChartConfig } from '@meldui/charts-vue'

const config: MeldAreaChartConfig = {
  series: [
    { name: 'Email', data: [120, 132, 101, 134, 90, 230] },
    { name: 'Push', data: [220, 182, 191, 234, 290, 330] },
    { name: 'In-App', data: [150, 232, 201, 154, 190, 330] },
  ],
  xAxis: {
    categories: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'],
  },
  stacked: true,
  fillOpacity: 0.6,
  stroke: { curve: 'smooth', width: 2 },
  tooltip: { shared: true },
}
</script>

<template>
  <MeldAreaChart title="Notifications by Channel" :config="config" />
</template>
```

## Fill Opacity

Control the transparency of the filled region with `fillOpacity` (0 for fully transparent, 1 for fully opaque):

```vue
<script setup lang="ts">
import { MeldAreaChart } from '@meldui/charts-vue'
import type { MeldAreaChartConfig } from '@meldui/charts-vue'

const config: MeldAreaChartConfig = {
  series: [{ name: 'Revenue', data: [3100, 4000, 2800, 5100, 4200, 6500] }],
  xAxis: {
    categories: ['Q1', 'Q2', 'Q3', 'Q4', 'Q5', 'Q6'],
  },
  fillOpacity: 0.15,
  stroke: { curve: 'smooth', width: 2 },
}
</script>

<template>
  <MeldAreaChart title="Quarterly Revenue" :config="config" />
</template>
```

## Smooth Curves with Data Points

Combine `stroke.curve`, `showPoints`, and `fillOpacity` for a polished look:

```vue
<script setup lang="ts">
import { MeldAreaChart } from '@meldui/charts-vue'
import type { MeldAreaChartConfig } from '@meldui/charts-vue'

const config: MeldAreaChartConfig = {
  series: [
    { name: 'CPU', data: [45, 52, 38, 65, 72, 61, 55] },
    { name: 'Memory', data: [62, 58, 70, 68, 75, 72, 69] },
  ],
  xAxis: {
    categories: ['00:00', '04:00', '08:00', '12:00', '16:00', '20:00', '24:00'],
  },
  stroke: { curve: 'smooth', width: 2 },
  showPoints: true,
  pointSize: 4,
  fillOpacity: 0.2,
  legend: { show: true },
  tooltip: { shared: true },
}
</script>

<template>
  <MeldAreaChart title="Server Resource Usage" :config="config" :height="380" />
</template>
```

## Data Labels

Show values on each data point:

```vue
<script setup lang="ts">
import { MeldAreaChart } from '@meldui/charts-vue'
import type { MeldAreaChartConfig } from '@meldui/charts-vue'

const config: MeldAreaChartConfig = {
  series: [{ name: 'Growth', data: [12, 18, 24, 32, 38, 45] }],
  xAxis: {
    categories: ['Week 1', 'Week 2', 'Week 3', 'Week 4', 'Week 5', 'Week 6'],
  },
  dataLabels: {
    show: true,
    position: 'top',
    formatter: (value: number) => `${value}%`,
  },
  fillOpacity: 0.25,
}
</script>

<template>
  <MeldAreaChart title="User Growth Rate" :config="config" />
</template>
```

## Custom Colors

Override the default palette with per-series colors or a named palette:

```vue
<script setup lang="ts">
import { MeldAreaChart } from '@meldui/charts-vue'
import type { MeldAreaChartConfig } from '@meldui/charts-vue'

const config: MeldAreaChartConfig = {
  series: [
    { name: 'Income', data: [5000, 5500, 6000, 5800, 6200, 6800], color: '#10b981' },
    { name: 'Expenses', data: [3200, 3100, 3500, 3800, 3600, 3900], color: '#ef4444' },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  fillOpacity: 0.2,
  stroke: { curve: 'smooth', width: 2 },
}
</script>

<template>
  <MeldAreaChart title="Income vs Expenses" :config="config" />
</template>
```

## API Reference

### MeldAreaChartConfig

Extends `CartesianChartConfigBase`.

| Property      | Type                                | Default     | Description                             |
| ------------- | ----------------------------------- | ----------- | --------------------------------------- |
| `series`      | `ChartSeries[]`                     | _required_  | Data series to display.                 |
| `xAxis`       | `ChartAxis`                         | `undefined` | X-axis configuration.                   |
| `yAxis`       | `ChartAxis`                         | `undefined` | Y-axis configuration.                   |
| `stroke`      | `ChartStroke`                       | `undefined` | Line width, curve type, and dash style. |
| `stacked`     | `boolean`                           | `false`     | Stack areas on top of each other.       |
| `fillOpacity` | `number`                            | `undefined` | Opacity of the filled region (0 to 1).  |
| `showPoints`  | `boolean`                           | `false`     | Show markers at each data point.        |
| `pointSize`   | `number`                            | `undefined` | Marker radius in pixels.                |
| `dataLabels`  | `ChartDataLabels`                   | `undefined` | Show values on data points.             |
| `legend`      | `ChartLegend`                       | `undefined` | Legend visibility and position.         |
| `tooltip`     | `ChartTooltip`                      | `undefined` | Tooltip behavior and formatting.        |
| `grid`        | `ChartGrid`                         | `undefined` | Grid lines configuration.               |
| `colors`      | `'auto' \| PaletteName \| string[]` | `'auto'`    | Series colors.                          |
| `animations`  | `boolean`                           | `true`      | Enable or disable animations.           |
| `toolbar`     | `boolean`                           | `false`     | Show toolbar.                           |
| `zoom`        | `boolean`                           | `false`     | Enable zoom.                            |
| `advanced`    | `Record<string, unknown>`           | `undefined` | Raw ECharts options for edge cases.     |
