## Basic Usage

The `MeldScatterChart` component plots individual data points on a two-dimensional grid. It is ideal for showing correlations, distributions, and clusters in your data.

For scatter charts, data is typically provided as `[x, y]` pairs:

```vue
<script setup lang="ts">
import { MeldScatterChart } from '@meldui/charts-vue'
import type { MeldScatterChartConfig } from '@meldui/charts-vue'

const config: MeldScatterChartConfig = {
  series: [
    {
      name: 'Measurements',
      data: [
        { x: 10, y: 8.04 },
        { x: 8, y: 6.95 },
        { x: 13, y: 7.58 },
        { x: 9, y: 8.81 },
        { x: 11, y: 8.33 },
        { x: 14, y: 9.96 },
        { x: 6, y: 7.24 },
        { x: 4, y: 4.26 },
        { x: 12, y: 10.84 },
        { x: 7, y: 4.82 },
        { x: 5, y: 5.68 },
      ],
    },
  ],
  xAxis: { title: 'X Value', type: 'numeric' },
  yAxis: { title: 'Y Value' },
}
</script>

<template>
  <MeldScatterChart title="Data Distribution" :config="config" />
</template>
```

## Multiple Series

Compare groups of data points by adding multiple series:

```vue
<script setup lang="ts">
import { MeldScatterChart } from '@meldui/charts-vue'
import type { MeldScatterChartConfig } from '@meldui/charts-vue'

const config: MeldScatterChartConfig = {
  series: [
    {
      name: 'Product A',
      data: [
        { x: 100, y: 4.5 },
        { x: 200, y: 3.8 },
        { x: 150, y: 4.2 },
        { x: 250, y: 3.5 },
        { x: 180, y: 4.0 },
        { x: 300, y: 3.2 },
        { x: 120, y: 4.3 },
        { x: 220, y: 3.9 },
      ],
    },
    {
      name: 'Product B',
      data: [
        { x: 80, y: 3.0 },
        { x: 160, y: 2.8 },
        { x: 240, y: 2.5 },
        { x: 130, y: 3.2 },
        { x: 190, y: 2.7 },
        { x: 280, y: 2.3 },
        { x: 110, y: 3.1 },
        { x: 210, y: 2.6 },
      ],
    },
  ],
  xAxis: { title: 'Price ($)', type: 'numeric' },
  yAxis: { title: 'Rating' },
  legend: { show: true, position: 'top' },
}
</script>

<template>
  <MeldScatterChart title="Price vs Rating" :config="config" :height="400" />
</template>
```

## Point Shape

Change the shape of data points with `pointShape`:

```vue
<script setup lang="ts">
import { MeldScatterChart } from '@meldui/charts-vue'
import type { MeldScatterChartConfig } from '@meldui/charts-vue'

const config: MeldScatterChartConfig = {
  series: [
    {
      name: 'Group A',
      data: [
        { x: 1, y: 3 },
        { x: 2, y: 5 },
        { x: 3, y: 4 },
        { x: 4, y: 7 },
        { x: 5, y: 6 },
      ],
    },
    {
      name: 'Group B',
      data: [
        { x: 1.5, y: 4 },
        { x: 2.5, y: 6 },
        { x: 3.5, y: 5 },
        { x: 4.5, y: 8 },
        { x: 5.5, y: 7 },
      ],
    },
  ],
  xAxis: { type: 'numeric' },
  pointShape: 'diamond',
  pointSize: 10,
  legend: { show: true },
}
</script>

<template>
  <MeldScatterChart title="Grouped Comparison" :config="config" />
</template>
```

Available point shapes:

| Value        | Description                |
| ------------ | -------------------------- |
| `'circle'`   | Circular points (default). |
| `'rect'`     | Square points.             |
| `'triangle'` | Triangular points.         |
| `'diamond'`  | Diamond-shaped points.     |

## Category-Based Scatter

Use category labels on the x-axis instead of numeric values:

```vue
<script setup lang="ts">
import { MeldScatterChart } from '@meldui/charts-vue'
import type { MeldScatterChartConfig } from '@meldui/charts-vue'

const config: MeldScatterChartConfig = {
  series: [
    { name: 'Team Alpha', data: [85, 72, 90, 68, 95] },
    { name: 'Team Beta', data: [78, 88, 65, 82, 70] },
  ],
  xAxis: {
    categories: ['Sprint 1', 'Sprint 2', 'Sprint 3', 'Sprint 4', 'Sprint 5'],
    title: 'Sprint',
  },
  yAxis: { title: 'Velocity' },
  pointSize: 8,
  legend: { show: true },
}
</script>

<template>
  <MeldScatterChart title="Sprint Velocity" :config="config" />
</template>
```

## Data Labels

Display values next to each point:

```vue
<script setup lang="ts">
import { MeldScatterChart } from '@meldui/charts-vue'
import type { MeldScatterChartConfig } from '@meldui/charts-vue'

const config: MeldScatterChartConfig = {
  series: [
    {
      name: 'Cities',
      data: [
        { x: 50, y: 8200 },
        { x: 120, y: 12500 },
        { x: 80, y: 9800 },
        { x: 200, y: 18000 },
        { x: 35, y: 6500 },
      ],
    },
  ],
  xAxis: { title: 'Population (thousands)', type: 'numeric' },
  yAxis: { title: 'Avg. Income ($)' },
  dataLabels: {
    show: true,
    position: 'top',
    formatter: (value: number) => `$${(value / 1000).toFixed(1)}k`,
  },
  pointSize: 10,
}
</script>

<template>
  <MeldScatterChart title="Population vs Income" :config="config" :height="400" />
</template>
```

## Zoom and Toolbar

Enable zoom for exploring dense scatter plots:

```vue
<script setup lang="ts">
import { MeldScatterChart } from '@meldui/charts-vue'
import type { MeldScatterChartConfig } from '@meldui/charts-vue'

// Generate random scatter data
function generateData(count: number) {
  return Array.from({ length: count }, () => ({
    x: Math.random() * 100,
    y: Math.random() * 100,
  }))
}

const config: MeldScatterChartConfig = {
  series: [
    { name: 'Cluster A', data: generateData(50) },
    { name: 'Cluster B', data: generateData(50) },
  ],
  xAxis: { type: 'numeric', title: 'X' },
  yAxis: { title: 'Y' },
  zoom: true,
  toolbar: true,
  pointSize: 6,
  legend: { show: true },
  colors: 'vibrant',
}
</script>

<template>
  <MeldScatterChart title="Cluster Analysis" :config="config" :height="450" />
</template>
```

## API Reference

### MeldScatterChartConfig

Extends `CartesianChartConfigBase`.

| Property     | Type                                            | Default     | Description                          |
| ------------ | ----------------------------------------------- | ----------- | ------------------------------------ |
| `series`     | `ChartSeries[]`                                 | _required_  | Data series to plot.                 |
| `xAxis`      | `ChartAxis`                                     | `undefined` | X-axis configuration.                |
| `yAxis`      | `ChartAxis`                                     | `undefined` | Y-axis configuration.                |
| `pointSize`  | `number`                                        | `undefined` | Fixed size of data points in pixels. |
| `pointShape` | `'circle' \| 'rect' \| 'triangle' \| 'diamond'` | `'circle'`  | Shape of data points.                |
| `dataLabels` | `ChartDataLabels`                               | `undefined` | Show values on data points.          |
| `legend`     | `ChartLegend`                                   | `undefined` | Legend visibility and position.      |
| `tooltip`    | `ChartTooltip`                                  | `undefined` | Tooltip behavior and formatting.     |
| `grid`       | `ChartGrid`                                     | `undefined` | Grid lines configuration.            |
| `colors`     | `'auto' \| PaletteName \| string[]`             | `'auto'`    | Series colors.                       |
| `animations` | `boolean`                                       | `true`      | Enable or disable animations.        |
| `toolbar`    | `boolean`                                       | `false`     | Show toolbar.                        |
| `zoom`       | `boolean`                                       | `false`     | Enable zoom.                         |
| `advanced`   | `Record<string, unknown>`                       | `undefined` | Raw ECharts options for edge cases.  |
