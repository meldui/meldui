## Basic Usage

The `MeldHeatmapChart` component visualizes matrix data as a grid of colored cells. The color intensity of each cell represents its value. Heatmaps are effective for spotting patterns in two-dimensional data such as activity over time, correlation matrices, or frequency distributions.

For heatmap charts, data is provided as `[x, y, value]` triples within a single series:

```vue
<script setup lang="ts">
import { MeldHeatmapChart } from '@meldui/charts-vue'
import type { MeldHeatmapChartConfig } from '@meldui/charts-vue'

const config: MeldHeatmapChartConfig = {
  series: [
    {
      name: 'Commits',
      data: [
        { x: 'Mon', y: '9am' },
        { x: 'Mon', y: '12pm' },
        { x: 'Mon', y: '3pm' },
        { x: 'Tue', y: '9am' },
        { x: 'Tue', y: '12pm' },
        { x: 'Tue', y: '3pm' },
        { x: 'Wed', y: '9am' },
        { x: 'Wed', y: '12pm' },
        { x: 'Wed', y: '3pm' },
        { x: 'Thu', y: '9am' },
        { x: 'Thu', y: '12pm' },
        { x: 'Thu', y: '3pm' },
        { x: 'Fri', y: '9am' },
        { x: 'Fri', y: '12pm' },
        { x: 'Fri', y: '3pm' },
      ].map((item, i) => ({
        ...item,
        y: [5, 12, 8, 3, 15, 10, 7, 18, 14, 6, 11, 9, 4, 16, 2][i],
      })),
    },
  ],
  xAxis: {
    categories: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri'],
    title: 'Day',
  },
  yAxis: {
    categories: ['9am', '12pm', '3pm'],
    title: 'Time',
  },
}
</script>

<template>
  <MeldHeatmapChart title="Commit Activity" :config="config" />
</template>
```

A simpler approach uses flat numeric arrays where each series represents a row:

```vue
<script setup lang="ts">
import { MeldHeatmapChart } from '@meldui/charts-vue'
import type { MeldHeatmapChartConfig } from '@meldui/charts-vue'

const config: MeldHeatmapChartConfig = {
  series: [
    { name: 'Morning', data: [5, 12, 8, 3, 15] },
    { name: 'Afternoon', data: [10, 7, 18, 14, 6] },
    { name: 'Evening', data: [2, 9, 4, 16, 11] },
  ],
  xAxis: {
    categories: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri'],
  },
}
</script>

<template>
  <MeldHeatmapChart title="Weekly Activity" :config="config" />
</template>
```

## Color Range

Customize the gradient color range using the `colorRange` property:

```vue
<script setup lang="ts">
import { MeldHeatmapChart } from '@meldui/charts-vue'
import type { MeldHeatmapChartConfig } from '@meldui/charts-vue'

const config: MeldHeatmapChartConfig = {
  series: [
    { name: 'Jan', data: [2, 5, 8, 12, 15, 18, 22, 25, 20, 14, 8, 3] },
    { name: 'Feb', data: [3, 6, 10, 14, 17, 20, 24, 27, 22, 16, 10, 5] },
    { name: 'Mar', data: [5, 8, 12, 16, 20, 24, 28, 30, 26, 19, 12, 7] },
  ],
  xAxis: {
    categories: [
      '6am',
      '7am',
      '8am',
      '9am',
      '10am',
      '11am',
      '12pm',
      '1pm',
      '2pm',
      '3pm',
      '4pm',
      '5pm',
    ],
    title: 'Hour',
  },
  colorRange: {
    min: 0,
    max: 30,
    colors: ['#eef2ff', '#818cf8', '#4338ca'], // Light indigo to dark indigo
  },
}
</script>

<template>
  <MeldHeatmapChart title="Temperature by Hour" :config="config" :height="300" />
</template>
```

## Show Values in Cells

Display numeric values inside each cell with `showValues`:

```vue
<script setup lang="ts">
import { MeldHeatmapChart } from '@meldui/charts-vue'
import type { MeldHeatmapChartConfig } from '@meldui/charts-vue'

const config: MeldHeatmapChartConfig = {
  series: [
    { name: 'JS', data: [1.0, 0.8, 0.6, 0.3] },
    { name: 'TS', data: [0.8, 1.0, 0.7, 0.4] },
    { name: 'Python', data: [0.6, 0.7, 1.0, 0.5] },
    { name: 'Go', data: [0.3, 0.4, 0.5, 1.0] },
  ],
  xAxis: {
    categories: ['JS', 'TS', 'Python', 'Go'],
  },
  showValues: true,
  colorRange: {
    min: 0,
    max: 1,
    colors: ['#fef3c7', '#f59e0b', '#b45309'],
  },
}
</script>

<template>
  <MeldHeatmapChart title="Language Similarity Matrix" :config="config" :height="350" />
</template>
```

## Cell Border

Add borders to heatmap cells for better visual separation:

```vue
<script setup lang="ts">
import { MeldHeatmapChart } from '@meldui/charts-vue'
import type { MeldHeatmapChartConfig } from '@meldui/charts-vue'

const config: MeldHeatmapChartConfig = {
  series: [
    { name: 'Week 1', data: [4, 7, 2, 8, 5, 3, 6] },
    { name: 'Week 2', data: [6, 3, 8, 5, 7, 4, 9] },
    { name: 'Week 3', data: [5, 8, 4, 6, 3, 7, 2] },
    { name: 'Week 4', data: [7, 2, 6, 9, 4, 8, 5] },
  ],
  xAxis: {
    categories: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'],
  },
  cellBorder: {
    show: true,
    color: '#ffffff',
    width: 2,
  },
  showValues: true,
  colorRange: {
    colors: ['#dcfce7', '#22c55e', '#15803d'],
  },
}
</script>

<template>
  <MeldHeatmapChart title="Daily Exercise (hours)" :config="config" :height="300" />
</template>
```

## Data Labels with Formatter

Use `dataLabels` with a custom formatter for display formatting:

```vue
<script setup lang="ts">
import { MeldHeatmapChart } from '@meldui/charts-vue'
import type { MeldHeatmapChartConfig } from '@meldui/charts-vue'

const config: MeldHeatmapChartConfig = {
  series: [
    { name: 'Q1', data: [85, 72, 90, 68] },
    { name: 'Q2', data: [78, 88, 65, 82] },
    { name: 'Q3', data: [92, 75, 85, 70] },
    { name: 'Q4', data: [80, 95, 78, 88] },
  ],
  xAxis: {
    categories: ['North', 'South', 'East', 'West'],
    title: 'Region',
  },
  dataLabels: {
    show: true,
    formatter: (value: number) => `${value}%`,
  },
  colorRange: {
    min: 60,
    max: 100,
    colors: ['#fecaca', '#ef4444', '#7f1d1d'],
  },
}
</script>

<template>
  <MeldHeatmapChart title="Sales Target Achievement" :config="config" :height="350" />
</template>
```

## API Reference

### MeldHeatmapChartConfig

Extends `CartesianChartConfigBase`.

| Property            | Type                                | Default     | Description                                                  |
| ------------------- | ----------------------------------- | ----------- | ------------------------------------------------------------ |
| `series`            | `ChartSeries[]`                     | _required_  | Data series representing rows or `[x, y, value]` triples.    |
| `xAxis`             | `ChartAxis`                         | `undefined` | X-axis configuration (column categories).                    |
| `yAxis`             | `ChartAxis`                         | `undefined` | Y-axis configuration (row categories).                       |
| `colorRange`        | `object`                            | `undefined` | Color gradient configuration.                                |
| `colorRange.min`    | `number`                            | `undefined` | Minimum value for color mapping.                             |
| `colorRange.max`    | `number`                            | `undefined` | Maximum value for color mapping.                             |
| `colorRange.colors` | `string[]`                          | `undefined` | Array of colors forming the gradient (min to max).           |
| `showValues`        | `boolean`                           | `false`     | Show numeric values inside each cell.                        |
| `cellBorder`        | `object`                            | `undefined` | Cell border configuration.                                   |
| `cellBorder.show`   | `boolean`                           | `false`     | Show borders around cells.                                   |
| `cellBorder.color`  | `string`                            | `undefined` | Border color.                                                |
| `cellBorder.width`  | `number`                            | `undefined` | Border width in pixels.                                      |
| `dataLabels`        | `ChartDataLabels`                   | `undefined` | Show formatted values in cells.                              |
| `legend`            | `ChartLegend`                       | `undefined` | Legend (visual map) visibility and position.                 |
| `tooltip`           | `ChartTooltip`                      | `undefined` | Tooltip behavior and formatting.                             |
| `grid`              | `ChartGrid`                         | `undefined` | Grid configuration.                                          |
| `colors`            | `'auto' \| PaletteName \| string[]` | `'auto'`    | Series colors (used as gradient if `colorRange` is not set). |
| `animations`        | `boolean`                           | `true`      | Enable or disable animations.                                |
| `toolbar`           | `boolean`                           | `false`     | Show toolbar.                                                |
| `zoom`              | `boolean`                           | `false`     | Enable zoom.                                                 |
| `advanced`          | `Record<string, unknown>`           | `undefined` | Raw ECharts options for edge cases.                          |
