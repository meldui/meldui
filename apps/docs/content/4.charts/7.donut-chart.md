## Basic Usage

The `MeldDonutChart` is a variation of the pie chart with a hollow center. The hole makes it easier to compare slice sizes and provides space for a summary label or KPI value in the center.

```vue
<script setup lang="ts">
import { MeldDonutChart } from '@meldui/charts-vue'
import type { MeldDonutChartConfig } from '@meldui/charts-vue'

const config: MeldDonutChartConfig = {
  series: [
    { name: 'Completed', data: [68] },
    { name: 'In Progress', data: [22] },
    { name: 'Pending', data: [10] },
  ],
}
</script>

<template>
  <MeldDonutChart title="Task Status" :config="config" />
</template>
```

## Center Label

Display a summary value or label in the center of the donut using the `centerLabel` property:

```vue
<script setup lang="ts">
import { MeldDonutChart } from '@meldui/charts-vue'
import type { MeldDonutChartConfig } from '@meldui/charts-vue'

const config: MeldDonutChartConfig = {
  series: [
    { name: 'Used', data: [73] },
    { name: 'Available', data: [27] },
  ],
  centerLabel: {
    show: true,
    title: 'Storage',
    value: '73%',
  },
  colors: ['#3b82f6', '#e2e8f0'],
}
</script>

<template>
  <MeldDonutChart title="Disk Usage" :config="config" />
</template>
```

## Inner Radius

Control the size of the hole using `innerRadius`. Larger values create a thinner ring:

```vue
<script setup lang="ts">
import { MeldDonutChart } from '@meldui/charts-vue'
import type { MeldDonutChartConfig } from '@meldui/charts-vue'

const config: MeldDonutChartConfig = {
  series: [
    { name: 'Active', data: [850] },
    { name: 'Inactive', data: [320] },
    { name: 'Suspended', data: [45] },
  ],
  innerRadius: '65%',
  radius: '80%',
  centerLabel: {
    show: true,
    title: 'Total Users',
    value: '1,215',
  },
  showLabels: true,
  labelPosition: 'outside',
}
</script>

<template>
  <MeldDonutChart title="User Status" :config="config" :height="400" />
</template>
```

## Labels and Legend

Combine outside labels with a positioned legend:

```vue
<script setup lang="ts">
import { MeldDonutChart } from '@meldui/charts-vue'
import type { MeldDonutChartConfig } from '@meldui/charts-vue'

const config: MeldDonutChartConfig = {
  series: [
    { name: 'Rent', data: [1500] },
    { name: 'Food', data: [600] },
    { name: 'Transport', data: [300] },
    { name: 'Utilities', data: [200] },
    { name: 'Entertainment', data: [250] },
    { name: 'Savings', data: [650] },
  ],
  showLabels: true,
  labelPosition: 'outside',
  legend: {
    show: true,
    position: 'bottom',
  },
  centerLabel: {
    show: true,
    title: 'Monthly',
    value: '$3,500',
  },
  colors: 'pastel',
}
</script>

<template>
  <MeldDonutChart title="Expense Breakdown" :config="config" :height="450" />
</template>
```

## Data Labels with Formatter

Format the values shown on slices:

```vue
<script setup lang="ts">
import { MeldDonutChart } from '@meldui/charts-vue'
import type { MeldDonutChartConfig } from '@meldui/charts-vue'

const config: MeldDonutChartConfig = {
  series: [
    { name: 'North America', data: [42000] },
    { name: 'Europe', data: [38000] },
    { name: 'Asia Pacific', data: [25000] },
    { name: 'Latin America', data: [15000] },
  ],
  dataLabels: {
    show: true,
    position: 'inside',
    formatter: (value: number) => `$${(value / 1000).toFixed(0)}k`,
  },
  innerRadius: '45%',
  centerLabel: {
    show: true,
    title: 'Total Revenue',
    value: '$120k',
  },
}
</script>

<template>
  <MeldDonutChart title="Revenue by Region" :config="config" :height="400" />
</template>
```

## Start Angle

Rotate the starting position of the first slice:

```vue
<script setup lang="ts">
import { MeldDonutChart } from '@meldui/charts-vue'
import type { MeldDonutChartConfig } from '@meldui/charts-vue'

const config: MeldDonutChartConfig = {
  series: [
    { name: 'Excellent', data: [42] },
    { name: 'Good', data: [31] },
    { name: 'Fair', data: [18] },
    { name: 'Poor', data: [9] },
  ],
  startAngle: 180,
  innerRadius: '50%',
  colors: 'earth',
}
</script>

<template>
  <MeldDonutChart title="Performance Ratings" :config="config" />
</template>
```

## API Reference

### MeldDonutChartConfig

Extends `ChartConfigBase` (no axis properties).

| Property            | Type                                | Default     | Description                                         |
| ------------------- | ----------------------------------- | ----------- | --------------------------------------------------- |
| `series`            | `ChartSeries[]`                     | _required_  | Data series (one entry per slice).                  |
| `showLabels`        | `boolean`                           | `false`     | Show labels on slices.                              |
| `labelPosition`     | `'inside' \| 'outside'`             | `undefined` | Where to place slice labels.                        |
| `startAngle`        | `number`                            | `undefined` | Starting angle in degrees.                          |
| `radius`            | `string \| number`                  | `undefined` | Outer radius as percentage or pixels.               |
| `innerRadius`       | `string \| number`                  | `undefined` | Inner radius that creates the hole (e.g., `'40%'`). |
| `centerLabel`       | `object`                            | `undefined` | Content displayed in the center of the donut.       |
| `centerLabel.show`  | `boolean`                           | `false`     | Show the center label.                              |
| `centerLabel.title` | `string`                            | `undefined` | Small title text above the value.                   |
| `centerLabel.value` | `string \| number`                  | `undefined` | Main value displayed in the center.                 |
| `dataLabels`        | `ChartDataLabels`                   | `undefined` | Show formatted values on slices.                    |
| `legend`            | `ChartLegend`                       | `undefined` | Legend visibility and position.                     |
| `tooltip`           | `ChartTooltip`                      | `undefined` | Tooltip behavior and formatting.                    |
| `colors`            | `'auto' \| PaletteName \| string[]` | `'auto'`    | Slice colors.                                       |
| `animations`        | `boolean`                           | `true`      | Enable or disable animations.                       |
| `advanced`          | `Record<string, unknown>`           | `undefined` | Raw ECharts options for edge cases.                 |
