## Basic Usage

The `MeldBarChart` component renders data as vertical bars by default. It is ideal for comparing discrete categories.

```vue
<script setup lang="ts">
import { MeldBarChart } from '@meldui/charts-vue'
import type { MeldBarChartConfig } from '@meldui/charts-vue'

const config: MeldBarChartConfig = {
  series: [{ name: 'Sales', data: [420, 380, 510, 590, 480, 620] }],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
}
</script>

<template>
  <MeldBarChart title="Monthly Sales" :config="config" />
</template>
```

## Grouped Bars

Add multiple series to create grouped (side-by-side) bars:

```vue
<script setup lang="ts">
import { MeldBarChart } from '@meldui/charts-vue'
import type { MeldBarChartConfig } from '@meldui/charts-vue'

const config: MeldBarChartConfig = {
  series: [
    { name: 'Revenue', data: [4200, 3800, 5100, 5900, 4800, 6200] },
    { name: 'Profit', data: [1200, 900, 1800, 2100, 1600, 2500] },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  legend: { show: true, position: 'top' },
  tooltip: { shared: true },
}
</script>

<template>
  <MeldBarChart title="Revenue vs Profit" :config="config" :height="400" />
</template>
```

## Stacked Bars

Set `stacked: true` to stack bars on top of each other:

```vue
<script setup lang="ts">
import { MeldBarChart } from '@meldui/charts-vue'
import type { MeldBarChartConfig } from '@meldui/charts-vue'

const config: MeldBarChartConfig = {
  series: [
    { name: 'Product A', data: [120, 132, 101, 134, 90, 230] },
    { name: 'Product B', data: [220, 182, 191, 234, 290, 330] },
    { name: 'Product C', data: [150, 232, 201, 154, 190, 330] },
  ],
  xAxis: {
    categories: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'],
  },
  stacked: true,
  tooltip: { shared: true },
  legend: { show: true },
}
</script>

<template>
  <MeldBarChart title="Daily Sales by Product" :config="config" />
</template>
```

## Horizontal Bars

Set `horizontal: true` to render bars horizontally. This is useful when category labels are long:

```vue
<script setup lang="ts">
import { MeldBarChart } from '@meldui/charts-vue'
import type { MeldBarChartConfig } from '@meldui/charts-vue'

const config: MeldBarChartConfig = {
  series: [{ name: 'Responses', data: [892, 756, 645, 534, 412, 387] }],
  xAxis: {
    categories: [
      'Very Satisfied',
      'Satisfied',
      'Neutral',
      'Dissatisfied',
      'Very Dissatisfied',
      'No Opinion',
    ],
  },
  horizontal: true,
  colors: 'ocean',
}
</script>

<template>
  <MeldBarChart title="Customer Satisfaction Survey" :config="config" :height="400" />
</template>
```

## Bar Width and Gap

Control the width of bars and the gap between them within a category:

```vue
<script setup lang="ts">
import { MeldBarChart } from '@meldui/charts-vue'
import type { MeldBarChartConfig } from '@meldui/charts-vue'

const config: MeldBarChartConfig = {
  series: [
    { name: '2024', data: [320, 410, 380, 520] },
    { name: '2025', data: [380, 470, 440, 590] },
  ],
  xAxis: {
    categories: ['Q1', 'Q2', 'Q3', 'Q4'],
  },
  barWidth: '40%',
  barGap: '20%',
}
</script>

<template>
  <MeldBarChart title="Year-over-Year Comparison" :config="config" />
</template>
```

## Data Labels

Display values directly on the bars:

```vue
<script setup lang="ts">
import { MeldBarChart } from '@meldui/charts-vue'
import type { MeldBarChartConfig } from '@meldui/charts-vue'

const config: MeldBarChartConfig = {
  series: [{ name: 'Budget', data: [85000, 72000, 63000, 51000, 45000] }],
  xAxis: {
    categories: ['Engineering', 'Marketing', 'Sales', 'Support', 'HR'],
  },
  dataLabels: {
    show: true,
    position: 'top',
    formatter: (value: number) => `$${(value / 1000).toFixed(0)}k`,
  },
}
</script>

<template>
  <MeldBarChart title="Department Budgets" :config="config" />
</template>
```

## Custom Series Colors

```vue
  xAxis: {
    categories: ['Q1', 'Q2', 'Q3', 'Q4'],
  },
  legend: { show: true },
}
</script>

<template>
  <MeldBarChart title="Target vs Actual" :config="config" />
</template>
```

## Handling Click Events

Respond to bar clicks:

```vue
<script setup lang="ts">
import { MeldBarChart } from '@meldui/charts-vue'
import type { MeldBarChartConfig, ChartClickEvent } from '@meldui/charts-vue'
import { ref } from 'vue'

const selected = ref('')

const config: MeldBarChartConfig = {
  series: [{ name: 'Votes', data: [342, 287, 198, 156, 134] }],
  xAxis: {
    categories: ['Feature A', 'Feature B', 'Feature C', 'Feature D', 'Feature E'],
  },
}

function handleClick(event: ChartClickEvent) {
  selected.value = `${event.name}: ${event.value} votes`
}
</script>

<template>
  <MeldBarChart title="Feature Requests" :config="config" @click="handleClick" />
  <p v-if="selected" class="mt-2 text-sm text-muted-foreground">{{ selected }}</p>
</template>
```

## API Reference

### MeldBarChartConfig

Extends `CartesianChartConfigBase`.

| Property     | Type                                | Default     | Description                                            |
| ------------ | ----------------------------------- | ----------- | ------------------------------------------------------ |
| `series`     | `ChartSeries[]`                     | _required_  | Data series to display.                                |
| `xAxis`      | `ChartAxis`                         | `undefined` | X-axis configuration.                                  |
| `yAxis`      | `ChartAxis`                         | `undefined` | Y-axis configuration.                                  |
| `horizontal` | `boolean`                           | `false`     | Render bars horizontally.                              |
| `stacked`    | `boolean`                           | `false`     | Stack bars on top of each other.                       |
| `dataLabels` | `ChartDataLabels`                   | `undefined` | Show values on bars.                                   |
| `barWidth`   | `number \| string`                  | `undefined` | Bar width in pixels or percentage (e.g., `'60%'`).     |
| `barGap`     | `string`                            | `undefined` | Gap between bars in the same category (e.g., `'30%'`). |
| `legend`     | `ChartLegend`                       | `undefined` | Legend visibility and position.                        |
| `tooltip`    | `ChartTooltip`                      | `undefined` | Tooltip behavior and formatting.                       |
| `grid`       | `ChartGrid`                         | `undefined` | Grid lines configuration.                              |
| `colors`     | `'auto' \| PaletteName \| string[]` | `'auto'`    | Series colors.                                         |
| `animations` | `boolean`                           | `true`      | Enable or disable animations.                          |
| `toolbar`    | `boolean`                           | `false`     | Show toolbar.                                          |
| `zoom`       | `boolean`                           | `false`     | Enable zoom.                                           |
| `advanced`   | `Record<string, unknown>`           | `undefined` | Raw ECharts options for edge cases.                    |
