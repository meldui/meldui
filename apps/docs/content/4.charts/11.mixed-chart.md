## Basic Usage

The `MeldMixedChart` component lets you combine line, bar, and area series in a single chart. Each series specifies its own `type` property. This is useful for dashboards where you need to show related metrics that are best represented with different visual encodings.

```vue
<script setup lang="ts">
import { MeldMixedChart } from '@meldui/charts-vue'
import type { MeldMixedChartConfig } from '@meldui/charts-vue'

const config: MeldMixedChartConfig = {
  series: [
    { name: 'Revenue', type: 'bar', data: [4200, 3800, 5100, 5900, 4800, 6200] },
    { name: 'Trend', type: 'line', data: [4000, 4100, 4500, 5000, 5200, 5800] },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  legend: { show: true },
}
</script>

<template>
  <MeldMixedChart title="Revenue with Trend Line" :config="config" />
</template>
```

## Bar + Line + Area

Mix all three types in a single visualization:

```vue
<script setup lang="ts">
import { MeldMixedChart } from '@meldui/charts-vue'
import type { MeldMixedChartConfig } from '@meldui/charts-vue'

const config: MeldMixedChartConfig = {
  series: [
    { name: 'Orders', type: 'bar', data: [320, 280, 410, 380, 450, 520] },
    { name: 'Revenue', type: 'line', data: [4200, 3800, 5100, 5900, 4800, 6200] },
    { name: 'Profit Margin', type: 'area', data: [28, 25, 32, 35, 30, 38] },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  legend: { show: true, position: 'top' },
  tooltip: { shared: true },
}
</script>

<template>
  <MeldMixedChart title="Sales Overview" :config="config" :height="400" />
</template>
```

## Dual Y-Axis

Use `yAxisIndex` on individual series and `yAxis2` to create a chart with two different scales. This is common when plotting metrics with very different value ranges (e.g., revenue in thousands alongside a percentage):

```vue
<script setup lang="ts">
import { MeldMixedChart } from '@meldui/charts-vue'
import type { MeldMixedChartConfig } from '@meldui/charts-vue'

const config: MeldMixedChartConfig = {
  series: [
    {
      name: 'Revenue ($)',
      type: 'bar',
      data: [42000, 38000, 51000, 59000, 48000, 62000],
      yAxisIndex: 0,
    },
    {
      name: 'Growth (%)',
      type: 'line',
      data: [12, 8, 18, 22, 15, 28],
      yAxisIndex: 1,
    },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  yAxis: {
    title: 'Revenue ($)',
  },
  yAxis2: {
    title: 'Growth (%)',
    min: 0,
    max: 40,
  },
  legend: { show: true },
  tooltip: { shared: true },
}
</script>

<template>
  <MeldMixedChart title="Revenue and Growth" :config="config" :height="400" />
</template>
```

## Stacked Mixed

Stack bars and areas while keeping lines unstacked:

```vue
<script setup lang="ts">
import { MeldMixedChart } from '@meldui/charts-vue'
import type { MeldMixedChartConfig } from '@meldui/charts-vue'

const config: MeldMixedChartConfig = {
  series: [
    { name: 'Product A', type: 'bar', data: [120, 132, 101, 134, 90, 230] },
    { name: 'Product B', type: 'bar', data: [220, 182, 191, 234, 290, 330] },
    { name: 'Total Target', type: 'line', data: [350, 350, 350, 350, 350, 350] },
  ],
  xAxis: {
    categories: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'],
  },
  stacked: true,
  legend: { show: true },
  tooltip: { shared: true },
}
</script>

<template>
  <MeldMixedChart title="Product Sales vs Target" :config="config" />
</template>
```

## Stroke Configuration

Apply stroke settings that affect line and area series:

```vue
<script setup lang="ts">
import { MeldMixedChart } from '@meldui/charts-vue'
import type { MeldMixedChartConfig } from '@meldui/charts-vue'

const config: MeldMixedChartConfig = {
  series: [
    { name: 'Actual', type: 'bar', data: [480, 520, 490, 560, 510, 580] },
    { name: 'Forecast', type: 'line', data: [500, 510, 520, 530, 540, 550] },
    { name: 'Lower Bound', type: 'area', data: [450, 460, 470, 480, 490, 500] },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  stroke: {
    curve: 'smooth',
    width: 2,
  },
  legend: { show: true },
}
</script>

<template>
  <MeldMixedChart title="Sales Forecast" :config="config" />
</template>
```

```vue

    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],

    position: 'top',
  },

</script>

<template>
  <MeldMixedChart title="Units and Pricing" :config="config" :height="400" />
</template>
```

```vue
  tooltip: { shared: true },
}
</script>

<template>
  <MeldMixedChart title="Financial Overview" :config="config" :height="400" />

```

## API Reference

### MeldMixedChartConfig

Extends `CartesianChartConfigBase`.

| Property     | Type                                | Default     | Description                            |
| ------------ | ----------------------------------- | ----------- | -------------------------------------- |
| `series`     | `MixedChartSeries[]`                | _required_  | Data series with type specification.   |
| `xAxis`      | `ChartAxis`                         | `undefined` | X-axis configuration.                  |
| `yAxis`      | `ChartAxis`                         | `undefined` | Primary Y-axis configuration.          |
| `yAxis2`     | `ChartAxis`                         | `undefined` | Secondary Y-axis for dual-axis charts. |
| `stroke`     | `ChartStroke`                       | `undefined` | Line/area stroke configuration.        |
| `stacked`    | `boolean`                           | `false`     | Stack bars and areas of the same type. |
| `dataLabels` | `ChartDataLabels`                   | `undefined` | Show values on chart elements.         |
| `legend`     | `ChartLegend`                       | `undefined` | Legend visibility and position.        |
| `tooltip`    | `ChartTooltip`                      | `undefined` | Tooltip behavior and formatting.       |
| `grid`       | `ChartGrid`                         | `undefined` | Grid lines configuration.              |
| `colors`     | `'auto' \| PaletteName \| string[]` | `'auto'`    | Series colors.                         |
| `animations` | `boolean`                           | `true`      | Enable or disable animations.          |
| `toolbar`    | `boolean`                           | `false`     | Show toolbar.                          |
| `zoom`       | `boolean`                           | `false`     | Enable zoom.                           |
| `advanced`   | `Record<string, unknown>`           | `undefined` | Raw ECharts options for edge cases.    |

### MixedChartSeries

Extends `ChartSeries`.

| Property     | Type                                                                    | Default     | Description                                       |
| ------------ | ----------------------------------------------------------------------- | ----------- | ------------------------------------------------- |
| `name`       | `string`                                                                | _required_  | Series name (shown in legend and tooltip).        |
| `data`       | `number[] \| Array<{ x: string \| number \| Date; y: number \| null }>` | _required_  | Data points.                                      |
| `type`       | `'line' \| 'bar' \| 'area'`                                             | _required_  | Chart type for this series.                       |
| `yAxisIndex` | `0 \| 1`                                                                | `0`         | Which Y-axis to use (0 = primary, 1 = secondary). |
| `color`      | `string`                                                                | `undefined` | Custom color for this series.                     |
