## Basic Usage

The `MeldPieChart` component displays data as slices of a circle. Each slice represents a series entry, and its size is proportional to the data value. Pie charts work best with a small number of categories (2-7).

For pie charts, each item in the `series` array represents one slice. Each series should have a single data point:

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'Chrome', data: [65] },
    { name: 'Firefox', data: [15] },
    { name: 'Safari', data: [12] },
    { name: 'Edge', data: [5] },
    { name: 'Other', data: [3] },
  ],
}
</script>

<template>
  <MeldPieChart title="Browser Market Share" :config="config" />
</template>
```

## Labels

Show labels on each slice using `showLabels` and control their placement with `labelPosition`:

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'Rent', data: [1200] },
    { name: 'Food', data: [400] },
    { name: 'Transport', data: [200] },
    { name: 'Entertainment', data: [150] },
    { name: 'Savings', data: [350] },
  ],
  showLabels: true,
  labelPosition: 'outside',
}
</script>

<template>
  <MeldPieChart title="Monthly Budget Breakdown" :config="config" :height="400" />
</template>
```

Available label positions:

| Value       | Description                                                            |
| ----------- | ---------------------------------------------------------------------- |
| `'inside'`  | Labels are rendered inside the slice. Best for large slices.           |
| `'outside'` | Labels are rendered outside with leader lines. Better for many slices. |

## Inside Labels

Place labels directly inside the slices for a cleaner look when slices are large enough:

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'Completed', data: [72] },
    { name: 'In Progress', data: [18] },
    { name: 'Not Started', data: [10] },
  ],
  showLabels: true,
  labelPosition: 'inside',
  colors: ['#10b981', '#f59e0b', '#94a3b8'],
}
</script>

<template>
  <MeldPieChart title="Project Status" :config="config" />
</template>
```

## Custom Radius

Adjust the pie size using the `radius` property:

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'Desktop', data: [55] },
    { name: 'Mobile', data: [35] },
    { name: 'Tablet', data: [10] },
  ],
  radius: '60%',
  showLabels: true,
  labelPosition: 'outside',
}
</script>

<template>
  <MeldPieChart title="Device Distribution" :config="config" />
</template>
```

## Start Angle

Rotate the starting position of the first slice using `startAngle` (in degrees):

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'Q1', data: [250] },
    { name: 'Q2', data: [310] },
    { name: 'Q3', data: [280] },
    { name: 'Q4', data: [360] },
  ],
  startAngle: 90,
  showLabels: true,
  labelPosition: 'outside',
}
</script>

<template>
  <MeldPieChart title="Quarterly Revenue" :config="config" />
</template>
```

## Custom Colors

Assign specific colors per slice using the `color` property on each series entry, or use a named palette:

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'Success', data: [842], color: '#10b981' },
    { name: 'Warning', data: [156], color: '#f59e0b' },
    { name: 'Error', data: [48], color: '#ef4444' },
    { name: 'Info', data: [234], color: '#3b82f6' },
  ],
  showLabels: true,
  labelPosition: 'outside',
}
</script>

<template>
  <MeldPieChart title="Alert Types (Last 24h)" :config="config" />
</template>
```

Or use a named palette for automatic color assignment:

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'Engineering', data: [35] },
    { name: 'Design', data: [20] },
    { name: 'Marketing', data: [18] },
    { name: 'Sales', data: [15] },
    { name: 'Support', data: [12] },
  ],
  colors: 'vibrant',
  showLabels: true,
}
</script>

<template>
  <MeldPieChart title="Team Size Distribution" :config="config" />
</template>
```

## Data Labels with Formatter

Use `dataLabels` with a custom `formatter` to display formatted values:

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'US', data: [45000] },
    { name: 'EU', data: [32000] },
    { name: 'APAC', data: [28000] },
    { name: 'LATAM', data: [12000] },
  ],
  dataLabels: {
    show: true,
    position: 'inside',
    formatter: (value: number) => `$${(value / 1000).toFixed(0)}k`,
  },
}
</script>

<template>
  <MeldPieChart title="Revenue by Region" :config="config" :height="400" />
</template>
```

## Legend Configuration

Control the legend's visibility and position:

```vue
<script setup lang="ts">
import { MeldPieChart } from '@meldui/charts-vue'
import type { MeldPieChartConfig } from '@meldui/charts-vue'

const config: MeldPieChartConfig = {
  series: [
    { name: 'JavaScript', data: [38] },
    { name: 'TypeScript', data: [28] },
    { name: 'Python', data: [22] },
    { name: 'Go', data: [12] },
  ],
  legend: {
    show: true,
    position: 'right',
    align: 'left',
  },
  colors: 'corporate',
}
</script>

<template>
  <MeldPieChart title="Language Popularity" :config="config" />
</template>
```

## API Reference

### MeldPieChartConfig

Extends `ChartConfigBase` (no axis properties).

| Property        | Type                                | Default     | Description                                         |
| --------------- | ----------------------------------- | ----------- | --------------------------------------------------- |
| `series`        | `ChartSeries[]`                     | _required_  | Data series (one entry per slice).                  |
| `showLabels`    | `boolean`                           | `false`     | Show labels on slices.                              |
| `labelPosition` | `'inside' \| 'outside'`             | `undefined` | Where to place slice labels.                        |
| `startAngle`    | `number`                            | `undefined` | Starting angle in degrees.                          |
| `radius`        | `string \| number`                  | `undefined` | Pie radius as percentage (e.g., `'75%'`) or pixels. |
| `dataLabels`    | `ChartDataLabels`                   | `undefined` | Show formatted values on slices.                    |
| `legend`        | `ChartLegend`                       | `undefined` | Legend visibility and position.                     |
| `tooltip`       | `ChartTooltip`                      | `undefined` | Tooltip behavior and formatting.                    |
| `colors`        | `'auto' \| PaletteName \| string[]` | `'auto'`    | Slice colors.                                       |
| `animations`    | `boolean`                           | `true`      | Enable or disable animations.                       |
| `advanced`      | `Record<string, unknown>`           | `undefined` | Raw ECharts options for edge cases.                 |
