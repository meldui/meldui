## Basic Usage

The `MeldLineChart` component renders one or more data series as lines on a cartesian grid. It is ideal for displaying trends, time series, and continuous data.

```vue
<script setup lang="ts">
import { MeldLineChart } from '@meldui/charts-vue'
import type { MeldLineChartConfig } from '@meldui/charts-vue'

const config: MeldLineChartConfig = {
  series: [{ name: 'Page Views', data: [1200, 1900, 1500, 2800, 2200, 3100] }],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
    title: 'Month',
  },
  yAxis: {
    title: 'Views',
  },
}
</script>

<template>
  <MeldLineChart title="Monthly Page Views" :config="config" />
</template>
```

## Multiple Series

Pass multiple objects in the `series` array to compare datasets:

```vue
<script setup lang="ts">
import { MeldLineChart } from '@meldui/charts-vue'
import type { MeldLineChartConfig } from '@meldui/charts-vue'

const config: MeldLineChartConfig = {
  series: [
    { name: 'Desktop', data: [4200, 3800, 5100, 5900, 4800, 6200] },
    { name: 'Mobile', data: [2800, 3200, 3400, 4100, 4500, 5100] },
    { name: 'Tablet', data: [800, 900, 1100, 1300, 1200, 1400] },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  legend: { show: true, position: 'top' },
  tooltip: { shared: true },
}
</script>

<template>
  <MeldLineChart title="Traffic by Device" :config="config" :height="400" />
</template>
```

## Smooth Curves

Use the `stroke.curve` property to control the line interpolation:

```vue
<script setup lang="ts">
import { MeldLineChart } from '@meldui/charts-vue'
import type { MeldLineChartConfig } from '@meldui/charts-vue'

const config: MeldLineChartConfig = {
  series: [{ name: 'Temperature', data: [5, 8, 14, 20, 25, 28, 30, 27, 22, 15, 9, 4] }],
  xAxis: {
    categories: [
      'Jan',
      'Feb',
      'Mar',
      'Apr',
      'May',
      'Jun',
      'Jul',
      'Aug',
      'Sep',
      'Oct',
      'Nov',
      'Dec',
    ],
  },
  stroke: {
    curve: 'smooth',
    width: 3,
  },
}
</script>

<template>
  <MeldLineChart title="Average Temperature" :config="config" />
</template>
```

Available curve types:

| Value        | Description                                       |
| ------------ | ------------------------------------------------- |
| `'straight'` | Straight line segments between points (default).  |
| `'smooth'`   | Smoothed Bezier curves.                           |
| `'stepline'` | Step-wise transitions (horizontal then vertical). |

## Data Points

Show markers at each data point by setting `showPoints` and optionally `pointSize`:

```vue
<script setup lang="ts">
import { MeldLineChart } from '@meldui/charts-vue'
import type { MeldLineChartConfig } from '@meldui/charts-vue'

const config: MeldLineChartConfig = {
  series: [{ name: 'Conversions', data: [45, 52, 38, 65, 72, 61, 80] }],
  xAxis: {
    categories: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'],
  },
  showPoints: true,
  pointSize: 6,
  stroke: { curve: 'smooth', width: 2 },
}
</script>

<template>
  <MeldLineChart title="Weekly Conversions" :config="config" />
</template>
```

## Dashed Lines

Apply dashed or dotted line styles using `stroke.dashArray`:

```vue
<script setup lang="ts">
import { MeldLineChart } from '@meldui/charts-vue'
import type { MeldLineChartConfig } from '@meldui/charts-vue'

const config: MeldLineChartConfig = {
  series: [
    { name: 'Actual', data: [100, 120, 115, 134, 168, 180] },
    { name: 'Target', data: [110, 120, 130, 140, 150, 160] },
  ],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  stroke: {
    width: 2,
    dashArray: [0, 5], // First series solid, second series dashed
  },
}
</script>

<template>
  <MeldLineChart title="Actual vs Target" :config="config" />
</template>
```


## Handling Events

Listen for click events on data points:

```vue
<script setup lang="ts">
import { MeldLineChart } from '@meldui/charts-vue'
import type { MeldLineChartConfig, ChartClickEvent } from '@meldui/charts-vue'

const config: MeldLineChartConfig = {
  series: [{ name: 'Revenue', data: [4200, 3800, 5100, 5900, 4800, 6200] }],
  xAxis: {
    categories: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
  },
  showPoints: true,
}

function handleClick(event: ChartClickEvent) {
  console.log(`Clicked ${event.seriesName} at ${event.name}: ${event.value}`)
}
</script>

<template>
  <MeldLineChart title="Click a data point" :config="config" @click="handleClick" />
</template>
```

## Zoom and Toolbar

Enable zoom and the built-in toolbar for interactive exploration:

```vue
<script setup lang="ts">
import { MeldLineChart } from '@meldui/charts-vue'
import type { MeldLineChartConfig } from '@meldui/charts-vue'

const config: MeldLineChartConfig = {
  series: [
    {
      name: 'CPU Usage',
      data: Array.from({ length: 100 }, () => Math.floor(Math.random() * 80 + 10)),
    },
  ],
  xAxis: {
    categories: Array.from({ length: 100 }, (_, i) => `${i}s`),
  },
  zoom: true,
  toolbar: true,
  stroke: { curve: 'smooth', width: 1.5 },
}
</script>

<template>
  <MeldLineChart title="CPU Usage Over Time" :config="config" :height="400" />
</template>
```

## API Reference

### MeldLineChartConfig

Extends `CartesianChartConfigBase`.

| Property     | Type                                | Default     | Description                                       |
| ------------ | ----------------------------------- | ----------- | ------------------------------------------------- |
| `series`     | `ChartSeries[]`                     | _required_  | Data series to display.                           |
| `xAxis`      | `ChartAxis`                         | `undefined` | X-axis configuration (categories, title, labels). |
| `yAxis`      | `ChartAxis`                         | `undefined` | Y-axis configuration (title, min/max, labels).    |
| `stroke`     | `ChartStroke`                       | `undefined` | Line width, curve type, and dash style.           |
| `showPoints` | `boolean`                           | `false`     | Show markers at each data point.                  |
| `pointSize`  | `number`                            | `undefined` | Marker radius in pixels.                          |
| `dataLabels` | `ChartDataLabels`                   | `undefined` | Show values on data points.                       |
| `legend`     | `ChartLegend`                       | `undefined` | Legend visibility and position.                   |
| `tooltip`    | `ChartTooltip`                      | `undefined` | Tooltip behavior and formatting.                  |
| `grid`       | `ChartGrid`                         | `undefined` | Grid lines configuration.                         |
| `colors`     | `'auto' \| PaletteName \| string[]` | `'auto'`    | Series colors.                                    |
| `animations` | `boolean`                           | `true`      | Enable or disable animations.                     |
| `toolbar`    | `boolean`                           | `false`     | Show toolbar with zoom and save tools.            |
| `zoom`       | `boolean`                           | `false`     | Enable zoom on the chart.                         |
| `advanced`   | `Record<string, unknown>`           | `undefined` | Raw ECharts options for edge cases.               |

### ChartStroke

| Property    | Type                                   | Default      | Description                                                               |
| ----------- | -------------------------------------- | ------------ | ------------------------------------------------------------------------- |
| `width`     | `number`                               | `2`          | Line thickness in pixels.                                                 |
| `curve`     | `'smooth' \| 'straight' \| 'stepline'` | `'straight'` | Line interpolation method.                                                |
| `dashArray` | `number \| number[]`                   | `undefined`  | Dash pattern. Single number applies to all series; array sets per-series. |

### ChartSeries

| Property | Type                                                                    | Default     | Description                                |
| -------- | ----------------------------------------------------------------------- | ----------- | ------------------------------------------ |
| `name`   | `string`                                                                | _required_  | Series name (shown in legend and tooltip). |
| `data`   | `number[] \| Array<{ x: string \| number \| Date; y: number \| null }>` | _required_  | Data points.                               |
| `color`  | `string`                                                                | `undefined` | Custom color for this series.              |
