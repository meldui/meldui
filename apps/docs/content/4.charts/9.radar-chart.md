## Basic Usage

The `MeldRadarChart` component (also known as a spider chart) displays multivariate data on a radial grid. Each axis represents a different variable, and data points are connected to form a polygon. Radar charts are useful for comparing entities across several dimensions.

```vue
<script setup lang="ts">
import { MeldRadarChart } from '@meldui/charts-vue'
import type { MeldRadarChartConfig } from '@meldui/charts-vue'

const config: MeldRadarChartConfig = {
  series: [{ name: 'Team Score', data: [85, 72, 90, 68, 78, 88] }],
  indicators: [
    { name: 'Communication', max: 100 },
    { name: 'Technical', max: 100 },
    { name: 'Creativity', max: 100 },
    { name: 'Leadership', max: 100 },
    { name: 'Teamwork', max: 100 },
    { name: 'Problem Solving', max: 100 },
  ],
}
</script>

<template>
  <MeldRadarChart title="Team Assessment" :config="config" />
</template>
```

## Multiple Series

Compare two or more entities on the same radar:

```vue
<script setup lang="ts">
import { MeldRadarChart } from '@meldui/charts-vue'
import type { MeldRadarChartConfig } from '@meldui/charts-vue'

const config: MeldRadarChartConfig = {
  series: [
    { name: 'Product A', data: [90, 75, 85, 60, 70] },
    { name: 'Product B', data: [65, 90, 70, 85, 80] },
  ],
  indicators: [
    { name: 'Performance', max: 100 },
    { name: 'Reliability', max: 100 },
    { name: 'Usability', max: 100 },
    { name: 'Price Value', max: 100 },
    { name: 'Support', max: 100 },
  ],
  legend: { show: true, position: 'top' },
}
</script>

<template>
  <MeldRadarChart title="Product Comparison" :config="config" :height="400" />
</template>
```

## Filled Area

Set `fillArea: true` to fill the polygon area, making it easier to compare the overall "footprint" of each series:

```vue
<script setup lang="ts">
import { MeldRadarChart } from '@meldui/charts-vue'
import type { MeldRadarChartConfig } from '@meldui/charts-vue'

const config: MeldRadarChartConfig = {
  series: [
    { name: 'Current Quarter', data: [88, 72, 95, 65, 80, 70] },
    { name: 'Previous Quarter', data: [75, 80, 78, 72, 68, 82] },
  ],
  indicators: [
    { name: 'Revenue' },
    { name: 'Customers' },
    { name: 'Satisfaction' },
    { name: 'Efficiency' },
    { name: 'Growth' },
    { name: 'Retention' },
  ],
  fillArea: true,
  fillOpacity: 0.25,
  legend: { show: true },
}
</script>

<template>
  <MeldRadarChart title="Quarterly Comparison" :config="config" />
</template>
```

## Shape: Circle vs Polygon

Control the shape of the radar grid with the `shape` property:

```vue
<script setup lang="ts">
import { MeldRadarChart } from '@meldui/charts-vue'
import type { MeldRadarChartConfig } from '@meldui/charts-vue'

const config: MeldRadarChartConfig = {
  series: [{ name: 'Skills', data: [90, 65, 85, 70, 80] }],
  indicators: [
    { name: 'HTML/CSS', max: 100 },
    { name: 'JavaScript', max: 100 },
    { name: 'Vue', max: 100 },
    { name: 'Node.js', max: 100 },
    { name: 'SQL', max: 100 },
  ],
  shape: 'circle',
  fillArea: true,
  fillOpacity: 0.2,
}
</script>

<template>
  <MeldRadarChart title="Developer Skills (Circle Grid)" :config="config" />
</template>
```

| Value       | Description                                                          |
| ----------- | -------------------------------------------------------------------- |
| `'polygon'` | Polygonal grid lines (default). Traditional spider chart appearance. |
| `'circle'`  | Circular grid lines. Softer, more modern look.                       |

## Custom Indicator Max Values

Each indicator can have its own maximum value, which is useful when variables are measured on different scales:

```vue
<script setup lang="ts">
import { MeldRadarChart } from '@meldui/charts-vue'
import type { MeldRadarChartConfig } from '@meldui/charts-vue'

const config: MeldRadarChartConfig = {
  series: [
    { name: 'Server A', data: [8, 99.9, 250, 16, 64] },
    { name: 'Server B', data: [4, 99.5, 500, 32, 128] },
  ],
  indicators: [
    { name: 'CPU Cores', max: 16 },
    { name: 'Uptime %', max: 100 },
    { name: 'Latency (ms)', max: 1000 },
    { name: 'RAM (GB)', max: 64 },
    { name: 'Storage (GB)', max: 256 },
  ],
  legend: { show: true },
  fillArea: true,
  fillOpacity: 0.15,
}
</script>

<template>
  <MeldRadarChart title="Server Comparison" :config="config" :height="420" />
</template>
```

## Show Axis Labels

Toggle the visibility of axis labels (indicator names) with `showAxisLabels`:

```vue
<script setup lang="ts">
import { MeldRadarChart } from '@meldui/charts-vue'
import type { MeldRadarChartConfig } from '@meldui/charts-vue'

const config: MeldRadarChartConfig = {
  series: [{ name: 'Rating', data: [4.5, 3.8, 4.2, 4.0, 3.5] }],
  indicators: [
    { name: 'Quality', max: 5 },
    { name: 'Speed', max: 5 },
    { name: 'Value', max: 5 },
    { name: 'Service', max: 5 },
    { name: 'Ambiance', max: 5 },
  ],
  showAxisLabels: true,
  shape: 'polygon',
  fillArea: true,
  fillOpacity: 0.3,
  colors: 'sunset',
}
</script>

<template>
  <MeldRadarChart title="Restaurant Review" :config="config" />
</template>
```

## API Reference

### MeldRadarChartConfig

Extends `ChartConfigBase`.

| Property         | Type                                    | Default     | Description                                    |
| ---------------- | --------------------------------------- | ----------- | ---------------------------------------------- |
| `series`         | `ChartSeries[]`                         | _required_  | Data series to display.                        |
| `indicators`     | `Array<{ name: string; max?: number }>` | `undefined` | Radar axes with names and optional max values. |
| `shape`          | `'polygon' \| 'circle'`                 | `'polygon'` | Shape of the radar grid.                       |
| `fillArea`       | `boolean`                               | `false`     | Fill the area inside the data polygon.         |
| `fillOpacity`    | `number`                                | `undefined` | Opacity of the filled area (0 to 1).           |
| `showAxisLabels` | `boolean`                               | `true`      | Show indicator name labels around the radar.   |
| `xAxis`          | `ChartAxis`                             | `undefined` | Used for indicator names (optional).           |
| `yAxis`          | `ChartAxis`                             | `undefined` | Used for max values (optional).                |
| `grid`           | `ChartGrid`                             | `undefined` | Grid/axis line visibility.                     |
| `legend`         | `ChartLegend`                           | `undefined` | Legend visibility and position.                |
| `tooltip`        | `ChartTooltip`                          | `undefined` | Tooltip behavior.                              |
| `colors`         | `'auto' \| PaletteName \| string[]`     | `'auto'`    | Series colors.                                 |
| `animations`     | `boolean`                               | `true`      | Enable or disable animations.                  |
| `advanced`       | `Record<string, unknown>`               | `undefined` | Raw ECharts options for edge cases.            |

### Indicator Object

| Property | Type     | Default     | Description                                                         |
| -------- | -------- | ----------- | ------------------------------------------------------------------- |
| `name`   | `string` | _required_  | Label shown on the radar axis.                                      |
| `max`    | `number` | `undefined` | Maximum value for this axis. If omitted, auto-calculated from data. |
