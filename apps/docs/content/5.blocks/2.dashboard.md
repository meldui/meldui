Dashboard blocks provide page-level patterns for building data-rich overview pages. They combine stat cards, charts, tables, and activity feeds into a responsive grid layout using MeldUI components.

## Page Layout

A typical dashboard page follows a three-section structure: a header with title and primary action, a stats grid, and a content area with charts and feeds side by side.

```vue
<script setup lang="ts">
import { IconPlus } from '@meldui/tabler-vue'
import { Button } from '@meldui/vue'

import StatsGrid from './StatsGrid.vue'
import ActivityFeed from './ActivityFeed.vue'
import OverviewChart from './OverviewChart.vue'
</script>

<template>
  <div class="space-y-6">
    <!-- Header -->
    <div class="flex items-center justify-between">
      <div>
        <h1 class="text-3xl font-bold tracking-tight">Dashboard</h1>
        <p class="text-muted-foreground">Overview of your workspace</p>
      </div>
      <Button>
        <IconPlus class="mr-2 h-4 w-4" />
        New Item
      </Button>
    </div>

    <!-- Stats -->
    <StatsGrid />

    <!-- Charts and Activity -->
    <div class="grid gap-4 md:grid-cols-2">
      <OverviewChart />
      <ActivityFeed />
    </div>
  </div>
</template>
```

## Stat Cards

Stat cards display key metrics at a glance. Each card shows a title, a prominent value, optional description text, an icon, and optional trend or progress indicators. Build a reusable `StatsCard` component and render a responsive grid of them.

```vue
<script setup lang="ts">
import { Card, CardContent, CardHeader, CardTitle, Progress } from '@meldui/vue'
import type { Component } from 'vue'

interface Props {
  title: string
  value: string | number
  description?: string
  icon: Component
  progress?: number
  trend?: {
    value: number
    isPositive: boolean
  }
}

defineProps<Props>()
</script>

<template>
  <Card>
    <CardHeader class="flex flex-row items-center justify-between space-y-0 pb-2">
      <CardTitle class="text-sm font-medium">{{ title }}</CardTitle>
      <component :is="icon" class="h-4 w-4 text-muted-foreground" />
    </CardHeader>
    <CardContent>
      <div class="text-2xl font-bold">{{ value }}</div>
      <p v-if="description" class="text-xs text-muted-foreground">
        {{ description }}
      </p>
      <div v-if="trend" class="flex items-center text-xs mt-1">
        <span :class="trend.isPositive ? 'text-green-600' : 'text-red-600'">
          {{ trend.isPositive ? '+' : '' }}{{ trend.value }}%
        </span>
        <span class="text-muted-foreground ml-1">from last period</span>
      </div>
      <Progress v-if="progress !== undefined" :model-value="progress" class="mt-2 h-1" />
    </CardContent>
  </Card>
</template>
```

Use it inside a responsive grid:

```vue
<script setup lang="ts">
import { IconChecklist, IconCircleCheck, IconProgress, IconUsers } from '@meldui/tabler-vue'
import { computed } from 'vue'

import StatsCard from './StatsCard.vue'

const stats = computed(() => [
  {
    title: 'Total Revenue',
    value: '$45,231',
    description: '+20.1% from last month',
    icon: IconChecklist,
  },
  {
    title: 'Active Users',
    value: '2,350',
    description: '+180 this week',
    icon: IconUsers,
    trend: { value: 8.2, isPositive: true },
  },
  {
    title: 'Completed',
    value: 142,
    description: '85% completion rate',
    icon: IconCircleCheck,
    progress: 85,
  },
  {
    title: 'In Progress',
    value: 28,
    description: '12 due this week',
    icon: IconProgress,
  },
])
</script>

<template>
  <div class="grid gap-4 md:grid-cols-2 lg:grid-cols-4">
    <StatsCard v-for="stat in stats" :key="stat.title" v-bind="stat" />
  </div>
</template>
```

## Activity Feed with Timeline

The Timeline composite component is ideal for showing recent events. Wrap it in a Card to match the dashboard style.

```vue
<script setup lang="ts">
import { IconCircleCheck, IconCircleDashed, IconProgress } from '@meldui/tabler-vue'
import {
  Badge,
  Card,
  CardContent,
  CardDescription,
  CardHeader,
  CardTitle,
  RelativeTime,
  Timeline,
  TimelineConnector,
  TimelineContent,
  TimelineDot,
  TimelineHeader,
  TimelineItem,
  TimelineSeparator,
  TimelineTitle,
} from '@meldui/vue'
import { ref } from 'vue'

const activities = ref([
  { id: 1, title: 'Deployed v2.1.0 to production', status: 'done', updatedAt: new Date() },
  {
    id: 2,
    title: 'Code review for auth module',
    status: 'in-progress',
    updatedAt: new Date(Date.now() - 3600000),
  },
  {
    id: 3,
    title: 'Design system audit',
    status: 'todo',
    updatedAt: new Date(Date.now() - 7200000),
  },
])

const getStatusIcon = (status: string) => {
  switch (status) {
    case 'done':
      return IconCircleCheck
    case 'in-progress':
      return IconProgress
    default:
      return IconCircleDashed
  }
}

const getDotClass = (status: string) => {
  switch (status) {
    case 'done':
      return 'bg-green-100 text-green-600'
    case 'in-progress':
      return 'bg-blue-100 text-blue-600'
    default:
      return 'bg-yellow-100 text-yellow-600'
  }
}
</script>

<template>
  <Card>
    <CardHeader>
      <CardTitle>Recent Activity</CardTitle>
      <CardDescription>Latest updates across your workspace</CardDescription>
    </CardHeader>
    <CardContent>
      <Timeline>
        <TimelineItem v-for="(item, index) in activities" :key="item.id">
          <TimelineSeparator>
            <TimelineDot size="lg" :class="`size-7 ${getDotClass(item.status)}`">
              <component :is="getStatusIcon(item.status)" class="size-4" />
            </TimelineDot>
            <TimelineConnector v-if="index < activities.length - 1" />
          </TimelineSeparator>
          <TimelineContent>
            <TimelineHeader>
              <TimelineTitle>{{ item.title }}</TimelineTitle>
              <span class="text-xs text-muted-foreground">
                <RelativeTime :date="item.updatedAt" />
              </span>
            </TimelineHeader>
            <Badge variant="secondary" :outline="true" class="mt-1 text-xs capitalize">
              {{ item.status }}
            </Badge>
          </TimelineContent>
        </TimelineItem>
      </Timeline>
    </CardContent>
  </Card>
</template>
```

## Data Table Section

For dashboards that display tabular data, use the DataTable composite inside a Card. This pattern works well for recent orders, user lists, or audit logs.

```vue
<script setup lang="ts">
import {
  Badge,
  Card,
  CardContent,
  CardDescription,
  CardHeader,
  CardTitle,
  DataTable,
  createColumnHelper,
  useDataTable,
} from '@meldui/vue'
import { h } from 'vue'

interface Order {
  id: string
  customer: string
  status: 'completed' | 'pending' | 'cancelled'
  amount: number
  date: string
}

const orders: Order[] = [
  { id: 'ORD-001', customer: 'Alice Chen', status: 'completed', amount: 250.0, date: '2025-04-10' },
  { id: 'ORD-002', customer: 'Bob Smith', status: 'pending', amount: 125.5, date: '2025-04-11' },
  {
    id: 'ORD-003',
    customer: 'Carol Jones',
    status: 'cancelled',
    amount: 89.99,
    date: '2025-04-11',
  },
]

const col = createColumnHelper<Order>()

const columns = [
  col.accessor('id', { header: 'Order' }),
  col.accessor('customer', { header: 'Customer' }),
  col.accessor('status', {
    header: 'Status',
    cell: ({ row }) => {
      const variant =
        row.original.status === 'completed'
          ? 'default'
          : row.original.status === 'pending'
            ? 'secondary'
            : 'destructive'
      return h(Badge, { variant }, () => row.original.status)
    },
  }),
  col.accessor('amount', {
    header: 'Amount',
    cell: ({ row }) => `$${row.original.amount.toFixed(2)}`,
  }),
]

const { table } = useDataTable({ data: orders, columns })
</script>

<template>
  <Card>
    <CardHeader>
      <CardTitle>Recent Orders</CardTitle>
      <CardDescription>Latest transactions this week</CardDescription>
    </CardHeader>
    <CardContent>
      <DataTable :table="table" />
    </CardContent>
  </Card>
</template>
```

## Design Guidance

**Grid layout.** Use a responsive CSS grid for the stats row (`grid-cols-2 lg:grid-cols-4`) and a two-column grid for the content area (`md:grid-cols-2`). This collapses gracefully on small screens.

**Consistent spacing.** Wrap the entire page in a container with `space-y-6` to keep vertical rhythm consistent between sections.

**Color coding.** Use semantic Tailwind colors (green for success, blue for in-progress, yellow/red for warnings) consistently across stat trends, badges, and timeline dots so users can scan status at a glance.

**Loading states.** Replace stat values with `Skeleton` components while data is being fetched. The same height and width as the real content avoids layout shift.
