---
title: Timeline
description: A vertical or horizontal timeline for displaying chronological events.
category: composites
order: 4
package: '@meldui/vue'
componentName: Timeline
subComponents:
  - name: Timeline
    description: Root container. Provides orientation and variant context
  - name: TimelineItem
    description: Wraps a single event (separator + content)
  - name: TimelineSeparator
    description: Container for the dot and connector
  - name: TimelineDot
    description: The circular indicator
    props:
      - name: size
        type: "'sm' | 'md' | 'lg'"
        description: Dot size
  - name: TimelineConnector
    description: The line between dots
    props:
      - name: thickness
        type: number
        description: Line thickness
  - name: TimelineContent
    description: The text/body area for an event
  - name: TimelineTitle
    description: Heading text for the event
  - name: TimelineDescription
    description: Supporting description text
  - name: TimelineTime
    description: Timestamp display
    props:
      - name: dateTime
        type: string
        description: ISO 8601 date string
props:
  - name: orientation
    type: "'vertical' | 'horizontal'"
    default: "'vertical'"
    description: The direction the timeline flows.
  - name: variant
    type: "'default' | 'outline'"
    default: "'default'"
    description: Visual variant of the timeline dots and connectors.
  - name: activeIndex
    type: number
    description: Index of the currently active item (zero-based). Items up to this index appear in an active state.
  - name: contentPosition
    type: "'right' | 'left' | 'alternate'"
    default: "'right'"
    description: Which side of the line the content appears on.
---
