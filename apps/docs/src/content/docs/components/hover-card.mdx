---
title: Hover Card
description: A rich content card that appears when hovering over a trigger element.
category: components
subcategory: Overlay
order: 56
package: '@meldui/vue'
componentName: HoverCard
subComponents:
  - name: HoverCardContent
    description: Rich content card.
    props:
      - name: side
        type: "'top' | 'right' | 'bottom' | 'left'"
        description: Which side of the trigger to position on.
      - name: sideOffset
        type: number
        default: '4'
        description: Distance from the trigger in pixels.
      - name: align
        type: "'start' | 'center' | 'end'"
        description: Alignment relative to trigger.
  - name: HoverCardTrigger
    description: Element that triggers the hover card.
