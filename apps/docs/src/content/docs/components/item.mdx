  - name: variant
    type: "'default' | 'outline' | 'muted'"
    default: "'default'"
    description: The visual style variant.
  - name: size
    type: "'default' | 'sm'"
    default: "'default'"
    description: The size of the item.
  - name: as
    type: string
    default: "'div'"
    description: The element to render as.
  - name: asChild
    type: boolean
    description: Render as child element.

## Usage

```vue
<script setup>
import {
  Item,
  ItemHeader,
  ItemMedia,
  ItemContent,
  ItemTitle,
  ItemDescription,
  ItemActions,
} from '@meldui/vue'
import { IconUser } from '@meldui/tabler-vue'
</script>

<template>
  <Item>
    <ItemMedia variant="icon">
      <IconUser class="size-5" />
    </ItemMedia>
    <ItemContent>
      <ItemTitle>John Doe</ItemTitle>
      <ItemDescription>Software Engineer</ItemDescription>
    </ItemContent>
    <ItemActions>
      <Button size="sm" variant="outline">View</Button>
    </ItemActions>
  </Item>
</template>
```

## Sub-components

| Component         | Description                                                        |
| ----------------- | ------------------------------------------------------------------ |
| `Item`            | Root container. Props: `variant`, `size`, `as`, `asChild`          |
| `ItemGroup`       | Groups multiple items                                              |
| `ItemHeader`      | Header section                                                     |
| `ItemMedia`       | Icon/image area. Props: `variant` ('default' \| 'icon' \| 'image') |
| `ItemContent`     | Text content area                                                  |
| `ItemTitle`       | Title text                                                         |
| `ItemDescription` | Description text                                                   |
| `ItemFooter`      | Footer section                                                     |
| `ItemActions`     | Action buttons area                                                |
| `ItemSeparator`   | Visual separator between items                                     |
