---
title: Select
description: A custom select component with keyboard navigation and grouping support.
category: components
subcategory: Form
order: 12
package: '@meldui/vue'
componentName: Select
subComponents:
  - name: SelectTrigger
    description: Button that opens the dropdown.
    props:
      - name: size
        type: "'sm' | 'default'"
        default: "'default'"
        description: Trigger button size.
  - name: SelectContent
    description: Dropdown panel containing options.
    props:
      - name: position
        type: "'popper'"
        default: "'popper'"
        description: Positioning strategy.
  - name: SelectItem
    description: Individual selectable option.
    props:
      - name: value
        type: string
        required: true
        description: The option value.
      - name: disabled
        type: boolean
        description: Disable this option.
  - name: SelectValue
    description: Displays the currently selected value.
    props:
      - name: placeholder
        type: string
        description: Text shown when no value is selected.
  - name: SelectGroup
    description: Groups related options together.
  - name: SelectLabel
    description: Label heading for a group.
  - name: SelectSeparator
    description: Visual divider between groups.
  - name: SelectScrollUpButton
    description: Scroll-up button for long option lists.
  - name: SelectScrollDownButton
    description: Scroll-down button for long option lists.
