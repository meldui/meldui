---
title: Popover
description: A floating content panel that appears on trigger click.
category: components
subcategory: Overlay
order: 54
package: '@meldui/vue'
componentName: Popover
subComponents:
  - name: PopoverContent
    description: Floating content container.
    props:
      - name: align
        type: "'start' | 'center' | 'end'"
        default: "'center'"
        description: Horizontal alignment relative to trigger.
      - name: side
        type: "'top' | 'right' | 'bottom' | 'left'"
        description: Which side of the trigger to position on.
      - name: sideOffset
        type: number
        default: '4'
        description: Distance from the trigger in pixels.
  - name: PopoverTrigger
    description: Element that opens the popover.
  - name: PopoverAnchor
    description: Custom anchor element for positioning.
