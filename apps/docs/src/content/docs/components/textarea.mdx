---
title: Textarea
description: A multi-line text input with optional auto-sizing support.
category: components
subcategory: Form
order: 11
package: '@meldui/vue'
componentName: Textarea
props:
  - name: modelValue
    type: string | number
    description: The textarea value. Use with v-model.
  - name: autosize
    type: boolean
    default: 'false'
    description: Automatically resize the textarea to fit content.
  - name: maxHeight
    type: string
    description: Maximum height when autosize is enabled (CSS value).
events:
  - name: update:modelValue
    payload: string | number
    description: Emitted when the value changes.
---
