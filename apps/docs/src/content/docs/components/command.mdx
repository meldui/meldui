---
title: Command
description: A command palette for searching and executing actions, similar to Cmd+K interfaces.
category: components
subcategory: Interactive
order: 5
package: '@meldui/vue'
componentName: Command
props:
  - name: modelValue
    type: string
    default: "''"
    description: The current search value. Use with v-model.
events:
  - name: update:modelValue
    payload: string
    description: Emitted when the search value changes.
  - name: highlight
    payload: '{ ref: HTMLElement, value: string } | undefined'
    description: Emitted when the highlighted element changes.
subComponents:
  - name: CommandDialog
    description: Command palette rendered in a modal dialog.
    props:
      - name: title
        type: string
        default: "'Command Palette'"
        description: Dialog title for accessibility.
      - name: description
        type: string
        default: "'Search for a command to run...'"
        description: Dialog description for accessibility.
  - name: CommandInput
    description: Search input with magnifying glass icon.
  - name: CommandList
    description: Scrollable list container for items and groups.
  - name: CommandGroup
    description: Groups related items under a heading.
    props:
      - name: heading
        type: string
        description: Group heading text.
  - name: CommandItem
    description: Selectable command item.
    props:
      - name: value
        type: string
        description: Value used for filtering and selection.
      - name: disabled
        type: boolean
        description: Disable this item.
  - name: CommandEmpty
    description: Shown when no items match the search query.
  - name: CommandSeparator
    description: Visual divider between groups.
  - name: CommandShortcut
    description: Displays a keyboard shortcut hint aligned to the right.
---
