---
title: Field
description: A form field layout component with label, content, description, and error display.
category: components
subcategory: Form
order: 19
package: '@meldui/vue'
componentName: Field
props:
  - name: orientation
    type: "'vertical' | 'horizontal' | 'responsive'"
    default: "'vertical'"
    description: The layout direction of the field.
subComponents:
  - name: FieldLabel
    description: Label text for the field.
  - name: FieldContent
    description: Wrapper for the input element.
  - name: FieldDescription
    description: Help text displayed below the input.
  - name: FieldError
    description: Error message display.
    props:
      - name: errors
        type: "Array<string | { message: string }>"
        description: Array of error messages to display.
  - name: FieldGroup
    description: Groups multiple fields together.
  - name: FieldSet
    description: Fieldset wrapper element.
  - name: FieldLegend
    description: Legend for a fieldset.
    props:
      - name: variant
        type: "'legend' | 'label'"
        description: Render as legend or styled label.
  - name: FieldTitle
    description: Section title within a field group.
  - name: FieldSeparator
    description: Visual separator between fields.
---
