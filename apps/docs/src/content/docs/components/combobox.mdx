---
title: Combobox
description: An autocomplete input with a filterable dropdown list for selecting values.
category: components
subcategory: Interactive
order: 7
package: '@meldui/vue'
componentName: Combobox
props:
  - name: modelValue
    type: 'T | Array<T>'
    description: The selected value. Use with v-model.
  - name: multiple
    type: boolean
    description: Whether multiple items can be selected.
  - name: open
    type: boolean
    description: Controlled open state. Use with v-model:open.
  - name: disabled
    type: boolean
    default: 'false'
    description: Whether the combobox is disabled.
events:
  - name: update:modelValue
    payload: T
    description: Emitted when the selected value changes.
  - name: update:open
    payload: boolean
    description: Emitted when the open state changes.
subComponents:
  - name: ComboboxAnchor
    description: Positions the dropdown relative to the input.
  - name: ComboboxInput
    description: Filterable text input.
    props:
      - name: displayValue
        type: "(val: any) => string"
        description: Function to format the selected value for display.
      - name: placeholder
        type: string
        description: Placeholder text when empty.
  - name: ComboboxTrigger
    description: Toggle button to open/close the dropdown.
  - name: ComboboxList
    description: Dropdown list container.
    props:
      - name: position
        type: "'popper' | 'item-aligned'"
        default: "'popper'"
        description: Positioning strategy.
      - name: align
        type: "'start' | 'center' | 'end'"
        default: "'center'"
        description: Horizontal alignment.
      - name: sideOffset
        type: number
        default: '4'
        description: Distance from anchor in pixels.
  - name: ComboboxItem
    description: Selectable option in the list.
    props:
      - name: value
        type: T
        required: true
        description: The option value.
      - name: disabled
        type: boolean
        description: Disable this option.
      - name: textValue
        type: string
        description: Text used for filtering when content is complex.
  - name: ComboboxItemIndicator
    description: Checkmark indicator for the selected item.
  - name: ComboboxGroup
    description: Groups related options under a heading.
    props:
      - name: heading
        type: string
        description: Group heading text.
  - name: ComboboxEmpty
    description: Shown when no options match the filter.
  - name: ComboboxSeparator
    description: Visual divider between groups.
  - name: ComboboxCancel
    description: Button to clear the selected value.
---
