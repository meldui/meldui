---
title: File Upload
description: A file upload component with drag and drop, previews, and progress tracking.
category: components
subcategory: Form
order: 26
package: '@meldui/vue'
componentName: FileUpload
props:
  - name: modelValue
    type: 'File[]'
    description: The list of selected files. Use with v-model.
  - name: accept
    type: string
    description: Accepted MIME types or file extensions.
  - name: maxFiles
    type: number
    default: Infinity
    description: Maximum number of files allowed.
  - name: maxSize
    type: number
    description: Maximum file size in bytes.
  - name: multiple
    type: boolean
    default: 'true'
    description: Whether multiple files can be selected.
  - name: disabled
    type: boolean
    default: 'false'
    description: Whether the upload is disabled.
events:
  - name: update:modelValue
    payload: 'File[]'
    description: Emitted when files change.
  - name: file-accept
    payload: File
    description: Emitted when a file passes validation.
  - name: file-reject
    payload: '{ file: File, errors: FileError[] }'
    description: Emitted when a file fails validation.
subComponents:
  - name: FileUploadDropzone
    description: Drag and drop target area.
    slots:
      - name: default
        props: "{ isDragging: boolean, isDisabled: boolean, openFilePicker: () => void }"
        description: Content with drag state access.
  - name: FileUploadTrigger
    description: Button that opens the file picker dialog.
  - name: FileUploadList
    description: Container for uploaded file items.
  - name: FileUploadItem
    description: Individual file display with status.
    props:
      - name: file
        type: File
        required: true
        description: The file object.
      - name: progress
        type: number
        description: Upload progress percentage (0-100).
      - name: status
        type: "'pending' | 'uploading' | 'complete' | 'error'"
        default: "'pending'"
        description: Upload status.
      - name: error
        type: string
        description: Error message for failed uploads.
  - name: FileUploadItemName
    description: Displays the file name.
  - name: FileUploadItemSize
    description: Displays the file size.
    props:
      - name: precision
        type: number
        default: '1'
        description: Decimal places for size display.
  - name: FileUploadItemPreview
    description: Image preview thumbnail.
  - name: FileUploadItemProgress
    description: Upload progress bar.
    props:
      - name: variant
        type: "'default' | 'success' | 'destructive'"
        default: "'default'"
        description: Progress bar color variant.
  - name: FileUploadItemDelete
    description: Button to remove the file.
  - name: FileUploadClear
    description: Button to clear all files.
---
