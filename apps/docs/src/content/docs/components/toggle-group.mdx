---
title: Toggle Group
description: A set of two-state buttons that can be toggled on or off, supporting single or multiple selection.
category: components
subcategory: Interactive
order: 4
package: '@meldui/vue'
componentName: ToggleGroup
subComponents:
  - name: ToggleGroupItem
    description: Individual toggle item
    props:
      - name: value
        type: AcceptableValue
        required: true
        description: Item value
      - name: disabled
        type: boolean
        description: Disable this item
      - name: variant
        type: "'default' | 'outline'"
        description: Visual style variant
      - name: size
        type: "'default' | 'sm' | 'lg'"
        description: Item size
props:
  - name: modelValue
    type: 'AcceptableValue | AcceptableValue[]'
    description: The controlled value. Use with v-model.
  - name: multiple
    type: boolean
    description: Whether multiple items can be selected at once.
  - name: variant
    type: "'default' | 'outline'"
    default: "'default'"
    description: The visual style variant for all items.
  - name: size
    type: "'default' | 'sm' | 'lg'"
    default: "'default'"
    description: The size of all items.
  - name: disabled
    type: boolean
    default: 'false'
    description: Whether the entire group is disabled.
  - name: orientation
    type: "'horizontal' | 'vertical'"
    description: The layout direction.
events:
  - name: update:modelValue
    payload: 'AcceptableValue | AcceptableValue[]'
    description: Emitted when the value changes.
