---
title: Tabs
description: Tabbed navigation with line and pill variants for organizing content.
category: components
subcategory: Layout
order: 32
package: '@meldui/vue'
componentName: Tabs
props:
  - name: variant
    type: "'line' | 'pill'"
    default: "'line'"
    description: The visual style of the tabs.
  - name: defaultValue
    type: string
    description: The default active tab value.
  - name: modelValue
    type: string
    description: The controlled active tab. Use with v-model.
subComponents:
  - name: TabsList
    description: Container for tab trigger buttons.
    props:
      - name: loop
        type: boolean
        description: Keyboard navigation loops from last to first tab.
  - name: TabsTrigger
    description: Button that activates a tab panel.
    props:
      - name: value
        type: string
        required: true
        description: Unique value associating this trigger with its content panel.
      - name: disabled
        type: boolean
        description: Prevents activation of this tab.
  - name: TabsContent
    description: Panel content shown when the associated trigger is active.
    props:
      - name: value
        type: string
        required: true
        description: Unique value associating this panel with its trigger.
      - name: forceMount
        type: boolean
        description: Force mount content for CSS animations.
---
